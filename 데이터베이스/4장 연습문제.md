## 소스 코드

# 1번
```
select ENAME '팀장이없는 사원' from emp
where mgr is null;
```
# 2번
```
select ename '사원명', (select dname from dept d where d.deptno = e.deptno) '부서명'
from emp e;
```
# 3번
```
select ename 'CHICAGO에 근무하는 사원' from emp e 
where exists (select * from dept d where e.DEPTNO = d.DEPTNO and d.LOC = 'CHICAGO');
```
# 4번
```
select ename '평균보다 급여가 많은 직원' from emp where SAL >= (select avg(sal) from emp); 
```
# 5번
```
select ename '자기부서 평균보다 급여가 많은 직원' from emp e
where sal > (select avg(sal) from emp where deptno = e.deptno);
```
